#include <iostream>
using namespace std;


int bil, i, j, angka, angka2;

int prima(int angka, int angka2)
{
   for(i=angka;i<=angka2;i++){
		bil=0;
		for(j=1;j<=i;j++){
			if(i%j==0){
				bil=bil+1;
			}
		}
		if (bil==2)
		cout<<i<<" ";
	}

   return i;
}

main()
{

	cout << "Menampilkan Deret Bilangan Prima 20 - 50"<<endl;
   cout << "Masukan angka 20 : "; cin>>angka;
   cout << "Masukan angka 50 : "; cin>>angka2;
	cout<<endl;

   cout << "Hasil bilangan prima dari 20 - 50 adalah : "<<endl;
   cout << " "<<prima(angka, angka2);


   return 0;
}
